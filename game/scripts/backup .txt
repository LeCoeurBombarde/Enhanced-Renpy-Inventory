    class Inventory:
        def __init__(self, slot_count=21):
            self.slots = [[] for _ in range(slot_count)]  # Each slot starts as an empty list
            self.max_items_per_slot = 99

        def add_item(self, item):
            for slot in self.slots:
                if slot and slot[0] == item and len(slot) < self.max_items_per_slot:
                    slot.append(item)
                    return

            # If item couldn't be stacked, find an empty slot to add it
            for slot in self.slots:
                if not slot:
                    slot.append(item)
                    return

            # No available slots to add the item
            return "Not enough space in the inventory"

        def remove_item(self, item):
                for slot in self.slots:
                    if slot and slot[0] == item:
                        slot.pop()
                        if not slot:
                            self.slots.remove(slot)
                        return

                return "Item not found in the inventory"


            

        def get_items(self):
            return self.items
